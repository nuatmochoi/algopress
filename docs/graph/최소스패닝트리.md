# 최소 스패닝 트리 - BOJ

[최소 스패닝 트리 - BOJ](https://www.acmicpc.net/problem/1197)

```py
def Find(x): # 부모가 같은지 비교
    if p[x] == x:
        return x
    else:
        y = Find(p[x])
        p[x] = y
        return y

def Union(x, y): #부모 병합
    x = Find(x)
    y = Find(y)
    if x != y:
        p[y] = x

V, E = map(int, input().split())
G = []
p = [i for i in range(V + 1)]
for _ in range(E):
    A, B, C = map(int, input().split())
    G.append([A, B, C])

G.sort(key=lambda x: x[2])

w, cnt = 0, 0
for i in range(E):
    s = G[i][0]
    e = G[i][1]
    d = G[i][2]
    if Find(s) != Find(e): # 사이클이 없는 경우(root 다른 경우)
        Union(s, e)
        w += d
        cnt += 1
    if cnt == V - 1:
        break
print(w)
```
